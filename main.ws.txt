rule ("Init Global") {
    event {
        Ongoing - Global;
    }
    actions {
        Disable Inspector Recording;
        Disable Built-In Game Mode Announcer;
        Disable Built-In Game Mode Completion;
        Disable Built-In Game Mode Scoring;
        Set Global Variable(RestartLobbyTime, Subtract(Add(Multiply(Workshop Setting Toggle(Custom String("Lobby Setting"), Custom String("Infinite Lobby Time"), False, 1), 999999999999), Workshop Setting Integer(Custom String("Lobby Setting"), Custom String("Restart Lobby After X minutes"), 15, 1, 60, 0)), True));
        Wait(True, Ignore Condition);
        Set Match Time(900);
        Start Forcing Spawn Room(All Teams, False);
        Start Forcing Spawn Room(All Teams, 1);
        Start Forcing Spawn Room(All Teams, 2);
        Stop Forcing Spawn Room(All Teams);
        Set Global Variable(EditorOn, Workshop Setting Toggle(Custom String("Editor"), Custom String("Editor Mode Enabled"), True, 0));
        If(Not(Global.EditorOn));
            "LeaderboardText = null\nLeaderboardRemake = null"
            Set Global Variable(LeaderboardEntries, Empty Array);
        End;
        Wait(0.512, Ignore Condition);
        If(Compare(Global.Checkpoints, ==, Null));
            Set Global Variable(Checkpoints, Empty Array);
    }
}

rule ("Init Player") {
    event {
        Player Joined Match;
        All;
        All;
    }
    conditions {
        Or(Not(Is Dummy Bot(Event Player)), Array Contains(Global.TestBotArray, Event Player)) == True;
    }
    actions {
        Disable Kill Feed(Event Player);
        Disable Game Mode HUD(Event Player);
        Disable Game Mode In-World UI(Event Player);
        Disable Messages(Event Player);
        Disable Movement Collision With Players(Event Player);
        "map variables need to load first"
        Wait Until(Compare(Total Time Elapsed, >, 0.528), 0.528);
        Wait Until(Has Spawned(Event Player), 999999999999);
        Wait(0.128, Ignore Condition);
        Call Subroutine(FullReset);
    }
}

rule ("Arrive at Checkpoint") {
    event {
        Ongoing - Each Player;
        All;
        All;
    }
    conditions {
        (Event Player).EditorCurrentMode == Null;
        (Event Player).CheckpointGoal != Null;
        Is Alive(Event Player) == True;
        "2.65"
        Distance Between((Event Player).CheckpointGoal, Update Every Frame(Eye Position(Event Player))) <= Global.RingSize;
    }
    actions {
        "$$ TODO: Map adjustable actual speed percent"
        Modify Player Variable(Event Player, Speed, Add, 1.5);
        "$$ TODO: Map adjustable actual speed percent"
        Set Move Speed(Event Player, (Event Player).Speed);
        Play Effect(Event Player, Buff Explosion Sound, Color(White), Eye Position(Event Player), 100);
        "$$ TODO Splits Logic Here"
        Set Player Variable(Event Player, TimeSplit, Total Time Elapsed);
        Modify Player Variable(Event Player, TimeSplit, Subtract, (Event Player).TimeStart);

        If(Compare((Event Player).CheckpointCurrentIdx, >=, 10));
            Set Player Variable(Event Player, TimeSplitString, String Slice((Event Player).TimeSplitString, Add(Index Of String Char((Event Player).TimeSplitString, Custom String("\n")), True), 999999999999));
        End;
        "$$ END Splits Logic Here"
        Modify Player Variable(Event Player, CheckpointCurrentIdx, Add, True);
        Call Subroutine(SetCheckpoints);
        If(Compare(Count Of(Value In Array(Global.Checkpoints, (Event Player).CheckpointCurrentIdx)), ==, 2));
            Teleport(Event Player, (Event Player).CheckpointCurrent);
        End;
        Abort If(Or(Compare(Count Of(Global.Checkpoints), <=, 1), Compare((Event Player).CheckpointCurrentIdx, <, Subtract(Count Of(Global.Checkpoints), True))));
        "Finished"
        Set Player Variable(Event Player, IsRacing, False);
        Set Player Variable(Event Player, TimeStop, Total Time Elapsed);
        Modify Player Variable(Event Player, TimeStop, Subtract, (Event Player).TimeStart);
        "Don't record editor times"
        Abort If(Global.EditorOn);
        "High precision adjustment"
        Modify Player Variable(Event Player, TimeStop, Subtract, 0.016);
        Modify Player Variable(Event Player, TimeStop, Add, Multiply(0.016, Max(False, Min(True, Divide(Mapped Array(Array(Multiply(2.0, Dot Product(Vector Towards(Last Of(Global.Checkpoints), (Event Player).PositionOld), Direction Towards((Event Player).PositionOld, (Event Player).PositionCurrent)))), Multiply(-0.5, Add(Current Array Element, Square Root(Subtract(Multiply(Current Array Element, Current Array Element), Multiply(4.0, Subtract(Dot Product(Vector Towards(Last Of(Global.Checkpoints), (Event Player).PositionOld), Vector Towards(Last Of(Global.Checkpoints), (Event Player).PositionOld)), Multiply(Global.RingSize, Global.RingSize)))))))), Distance Between((Event Player).PositionOld, (Event Player).PositionCurrent))))));
        If(Array Contains(Mapped Array(Global.LeaderboardEntries, First Of(Current Array Element)), Custom String("{0}", Event Player)));
            "you already have a time"
            If(Compare((Event Player).TimeStop, >, Value In Array(First Of(Filtered Array(Global.LeaderboardEntries, Compare(First Of(Current Array Element), ==, Custom String("{0}", Event Player)))), True)));
                Big Message(Event Player, Custom String("Completion time: {0}.{1}", Round To Integer((Event Player).TimeStop, Down), String Slice(First Of(Round To Integer(Multiply(Subtract(Add(True, (Event Player).TimeStop), Round To Integer((Event Player).TimeStop, Down)), 10000), Up)), 1, 4)));
                "new time worse"
                Abort;
            End;
            Big Message(Event Player, Custom String("New personal best: {0}.{1}", Round To Integer((Event Player).TimeStop, Down), String Slice(First Of(Round To Integer(Multiply(Subtract(Add(True, (Event Player).TimeStop), Round To Integer((Event Player).TimeStop, Down)), 10000), Up)), 1, 4)));
            Modify Global Variable(LeaderboardEntries, Remove From Array By Value, Filtered Array(Global.LeaderboardEntries, Compare(First Of(Current Array Element), ==, Custom String("{0}", Event Player))));
        "you are not in list yet"
        Else;
            Big Message(Event Player, Custom String("Completion time: {0}.{1}", Round To Integer((Event Player).TimeStop, Down), String Slice(First Of(Round To Integer(Multiply(Subtract(Add(True, (Event Player).TimeStop), Round To Integer((Event Player).TimeStop, Down)), 10000), Up)), 1, 4)));
            "board has room for more or faster than last entry"
            If(Or(Compare(Count Of(Global.LeaderboardEntries), <, 25), Compare((Event Player).TimeStop, <, Value In Array(Value In Array(Global.LeaderboardEntries, 24), True))));
                Modify Global Variable(LeaderboardEntries, Remove From Array By Index, 24);
            Else;
                "your time lower than last entry"
                Abort;
            End;
        End;
        Set Player Variable(Event Player, TimePersonalBest, Custom String("{0}.{1}", Round To Integer((Event Player).TimeStop, Down), String Slice(First Of(Round To Integer(Multiply(Subtract(Add(True, (Event Player).TimeStop), Round To Integer((Event Player).TimeStop, Down)), 10000), Up)), 1, 4)));
        Set Player Variable(Event Player, TimeSplitsPB, (Event Player).TimeSplits);
        "new top 5 if faster or entry doesnt exist yet"
        If(Or(Or(Or(Or(Or(Or(Or(Or(Or(Compare(Value In Array(First Of(Global.LeaderboardEntries), True), >, (Event Player).TimeStop), Not(First Of(Global.LeaderboardEntries))), Compare(Value In Array(Value In Array(Global.LeaderboardEntries, True), True), >, (Event Player).TimeStop)), Not(Value In Array(Global.LeaderboardEntries, True))), Compare(Value In Array(Value In Array(Global.LeaderboardEntries, 2), True), >, (Event Player).TimeStop)), Not(Value In Array(Global.LeaderboardEntries, 2))), Compare(Value In Array(Value In Array(Global.LeaderboardEntries, 3), True), >, (Event Player).TimeStop)), Not(Value In Array(Global.LeaderboardEntries, 3))), Compare(Value In Array(Value In Array(Global.LeaderboardEntries, 4), True), >, (Event Player).TimeStop)), Not(Value In Array(Global.LeaderboardEntries, 4))));
            Big Message(All Players(All Teams), Custom String("New top 5 time: {0}.{1} by {2}!", Round To Integer((Event Player).TimeStop, Down), String Slice(First Of(Round To Integer(Multiply(Subtract(Add(True, (Event Player).TimeStop), Round To Integer((Event Player).TimeStop, Down)), 10000), Up)), 1, 4), Event Player));
        End;
        Modify Global Variable(LeaderboardEntries, Append To Array, Array(Array(Custom String("{0}", Event Player), (Event Player).TimeStop)));
        Set Global Variable(LeaderboardRemake, True);
    }

        (Host Player).EditorCurrentMode == 3;
        Is Button Held(Host Player, Button(Melee)) == True;
        (Host Player).EditMeleeButtonPressed == False;
    }
    actions {
        Small Message(Host Player, Custom String("   {0} Checkpoint Inserted", Icon String(Halo)));
        Play Effect(Host Player, Buff Impact Sound, Color(White), Value In Array(Global.Checkpoints, (Host Player).CheckpointCurrentIdx), 100);
        Set Player Variable(Host Player, EditMeleeButtonPressed, True);
        Modify Global Variable(Checkpoints, Append To Array, Null);
        Set Global Variable(Checkpoints, Mapped Array(Global.Checkpoints, If-Then-Else(Compare(Current Array Index, <=, (Host Player).CheckpointCurrentIdx), Current Array Element, Value In Array(Global.Checkpoints, Subtract(Current Array Index, True)))));
        Modify Player Variable(Host Player, CheckpointCurrentIdx, Add, True);
        Call Subroutine(SetCheckpoints);
        Set Player Variable(Host Player, EditorCurrentMode, 2);
        Set Player Variable(Host Player, EditingActive, True);
        Call Subroutine(StartEditorCamera2);
        Set Player Variable(Host Player, EditCheckpointTemp, Value In Array(Global.Checkpoints, (Host Player).CheckpointCurrentIdx));
        Create Effect(Host Player, Good Aura, Color(Rose), (Host Player).EditCheckpointTemp, Subtract(Global.RingSize, 0.15), Visible To Position and Radius);
        Set Player Variable(Host Player, EditCheckpointEffect, Last Created Entity);
        Set Player Variable(Host Player, EditorIsMovingCheckpoint, True);
    }
}
