rule ("Init Global") {
    event {
        Ongoing - Global;
    }
    actions {
        Disable Inspector Recording;
        Disable Built-In Game Mode Announcer;
        Disable Built-In Game Mode Completion;
        Disable Built-In Game Mode Scoring;
        Set Global Variable(RestartLobbyTime, Subtract(Add(Multiply(Workshop Setting Toggle(Custom String("Lobby Setting"), Custom String("Infinite Lobby Time"), False, 1), 999999999999), Workshop Setting Integer(Custom String("Lobby Setting"), Custom String("Restart Lobby After X minutes"), 15, 1, 60, 0)), 1));
        Wait(1, Ignore Condition);
        Start Forcing Spawn Room(All Teams, 0);
        Start Forcing Spawn Room(All Teams, 1);
        Start Forcing Spawn Room(All Teams, 2);
        Stop Forcing Spawn Room(All Teams);
        Set Global Variable(LeaderboardText, Null);
        Set Global Variable(LeaderboardEntries, Empty Array);
        Set Global Variable(LeaderboardRemake, Null);
        Set Match Time(900);
        Wait(0.512, Ignore Condition);
        If(Compare(Global.Checkpoints, ==, 0));
            Set Global Variable(Checkpoints, Empty Array);
    }
}

rule ("Init Player") {
    event {
        Player Joined Match;
        All;
        All;
    }
    conditions {
        Or(Not(Is Dummy Bot(Event Player)), Array Contains(Global.TestBotArray, Event Player)) == True;
    }
    actions {
        Set Player Variable(Event Player, WsEditorOn, Workshop Setting Toggle(Custom String("Editor"), Custom String("Editor Mode Enabled"), True, 0));
        Disable Kill Feed(Event Player);
        Disable Game Mode HUD(Event Player);
        Disable Game Mode In-World UI(Event Player);
        Disable Messages(Event Player);
        Disable Movement Collision With Players(Event Player);
        Set Ability 1 Enabled(Event Player, False);
        Set Status(Event Player, Null, Invincible, 999999);
        "map variables need to load first"
        Wait Until(Compare(Total Time Elapsed, >, 0.528), 0.528);
        Wait(0.128, Ignore Condition);
        Call Subroutine(FullReset);
    }
}

    event {
        Ongoing - Global;
    }
    actions {
        "Made by FishoFire version 1.1\nwait to overwrite any from copy pastas"

        Modify Global Variable(MapVectorArray, Append To Array, If-Then-Else(Compare(Current Game Mode, ==, Game Mode(Capture The Flag)), Array(Flag Position(Team 1), Flag Position(Team 2)), If-Then-Else(Array Contains(Array(Game Mode(Team Deathmatch), Game Mode(Deathmatch)), Current Game Mode), First Of(Spawn Points(All Teams)), If-Then-Else(And(First Of(Payload Position), Not(Add(Objective Position(1), Objective Position(2)))), Payload Position, Array(Objective Position(0), Objective Position(1), Objective Position(2))))));

    conditions {

                Set Player Variable(Event Player, CurrentEditorMode, 2);
            End;

rule ("Editor | Disable Movement and Abilities") {
    event {
        Subroutine;
        DisableAbilities;
    }
    actions {
        Set Jump Enabled(Event Player, False);
        Set Crouch Enabled(Event Player, False);
        Set Primary Fire Enabled(Event Player, False);
        Set Secondary Fire Enabled(Event Player, False);
        Disallow Button(Event Player, Button(Secondary Fire));
        Set Ability 1 Enabled(Event Player, False);
        Set Ability 2 Enabled(Event Player, False);
        Set Move Speed(Event Player, 0);
    }
}

rule ("Editor | Enable Movement and Abilities") {
    event {
        Subroutine;
        EnableAbilities;
    }
    actions {
        Set Jump Enabled(Event Player, True);
        Set Crouch Enabled(Event Player, True);
        Set Primary Fire Enabled(Event Player, True);
        Set Secondary Fire Enabled(Event Player, True);
        Allow Button(Event Player, Button(Secondary Fire));
        "This stays off always"
        Set Ability 1 Enabled(Event Player, False);
        Set Ability 2 Enabled(Event Player, True);
        Set Move Speed(Event Player, 100);
    }
}
